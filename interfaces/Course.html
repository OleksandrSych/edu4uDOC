import { ObjectId } from 'mongodb';
import { Document } from 'mongoose';
import { Topic } from './topic.interface';

/**
 * Course mongoose Document
 */
export interface Course extends Document {
    /**
     * Course id
     */
    readonly _id: ObjectId;

    /**
     * Course Name
     */
    readonly courseName: string;

    /**
     * Course Description
     */
    readonly courseDescription: string;

    /**
     * Course parent Department id
     */
    parentDepartment: ObjectId;

    /**
     * Course teacher owner id
     */ 
    readonly teacherOwner: ObjectId;

    /**
     * Course teacher owner name
     */ 
    teacherName: string;

    /**
     * Course hours number
     */
    readonly hoursNumber: number;

    /**
     * Course lectures number
     */
    readonly lecturesNumber: number;

    /**
     * Course seminars number
     */
    readonly seminarsNumber: number;

    /**
     * Course practices number
     */
    readonly practicesNumber: number;

    /**
     * Course laboratory works number
     */
    readonly laboratoryWorksNumber: number;

    /**
     * Course self education hours
     */
    readonly selfEducationHours: number;

    /**
     * Course control works number
     */
    readonly controlWorksNumber: number;

    /**
     * Course consultations number
     */
    readonly consultationsNumber: number;

    /**
     * Course total works number
     */ 
    readonly worksNumber: number;

    /**
     * Course students id
     */
    readonly students: Array<ObjectId>;
    
    /**
     * Course mentors id
     */ 
    readonly mentors: Array<ObjectId>;

    /**
     * Course created Date
     */
    createdDate: Date;

    /**
     * Course created User id
     */
    createdUser: ObjectId; 

    /**
     * Invite key
     */
    inviteKey: ObjectId;   
    
    /**
     * Course topics
     */ 
    topics: Topic[]; 
}
