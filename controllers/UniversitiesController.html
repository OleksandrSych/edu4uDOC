import { Controller, UseGuards, Get, Res, Param, Post, Body, Put, Query, Delete, NotFoundException, HttpStatus } from '@nestjs/common';
import { JwtAuthGuard } from 'src/auth/jwt-auth.guard';
import { ApiTags, ApiBearerAuth, ApiUnauthorizedResponse, ApiOkResponse, ApiForbiddenResponse, ApiNotFoundResponse, ApiBadRequestResponse, ApiParam, ApiQuery } from '@nestjs/swagger';
import { ObjectID } from 'mongodb';
import { ValidateObjectId } from 'src/pipes/validate-object-id.pipes';
import { UniversitiesService } from './universities.service';
import { CreateNodeDTO } from './dto/create-node.dto';
import {UsersService} from '../users/users.service';
import {UsersModule} from '../users/users.module';

/**
 * Universities Controller
 */
@UseGuards(JwtAuthGuard)
@Controller('universities')
@ApiTags('Universities')
@ApiBearerAuth()
@ApiUnauthorizedResponse({description:'Access Token is not valid or has expired'})
@ApiForbiddenResponse({description:'Forbidden.'})
@ApiNotFoundResponse({description:'Node does not exist.'})
export class UniversitiesController {
      /**@ignore */
      constructor(private universitiesService: UniversitiesService ,private usersService:UsersService ) { }  
    
    /**
     * Fetch all tree of universities
     * @param res 
     */
    @Get('/tree')
    @ApiOkResponse({description:'The tree of universities list has been successfully returned.'})
    async getTree(@Res() res) {      
      var tree = await this.universitiesService.getTree();
      if (!tree) {
        throw new NotFoundException('Universities does not exist!');
      }
      return res.status(HttpStatus.OK).json(tree);
    }

    /**
     * Fetch all universities
     * @param res 
     */
    @Get('/universities')
    @ApiOkResponse({description:'The universities list has been successfully returned.'})
    async getNodes(@Res() res) {
      const Universities = await this.universitiesService.getNodes("");
      if (!Universities) {
        throw new NotFoundException('Universities does not exist!');
      }
      return res.status(HttpStatus.OK).json(Universities);
    }
       

    /**
     * Fetch a particular node using id
     * @param res 
     * @param id 
     */
    @Get(':id')
    @ApiOkResponse({ description:'The node has been successfully returned.'})    
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({ name: 'id', type: String })
    async getNode(@Res() res, @Param('id', new ValidateObjectId()) id) {
      const node = await this.universitiesService.getNode(new ObjectID(id));
      if (!node) {
          throw new NotFoundException('Node does not exist!');
      }
      return res.status(HttpStatus.OK).json(node);
    }
    
    /**
     * Fetch nodes using patent ID
     * @param res 
     * @param id 
     */
    @Get('nodes/:id')
    @ApiOkResponse({ description:'The nodes has been successfully returned by parent id'})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({ name: 'id', type: String })
    async getNodesPatentID(@Res() res, @Param('id', new ValidateObjectId()) id) {
      const nodes = await this.universitiesService.getNodes(id);
      if (!nodes) {
        throw new NotFoundException('nodes does not exist!');
      }
      return res.status(HttpStatus.OK).json(nodes);
    }

    /**
     * Submit a node
     * @param res 
     * @param createNodeDTO 
     */
    @Post('/add') 
    async addNode(@Res() res, @Body() createNodeDTO: CreateNodeDTO) { 
      const newNode = await this.universitiesService.addNode(createNodeDTO);
      if (!newNode) {
        throw new NotFoundException('newNode does not exist!');
      }
      return res.status(HttpStatus.OK).json({
        message: 'Node has been submitted successfully!',
        node: newNode,
      });
    }   

    /**
     * Edit a node
     * @param res 
     * @param id 
     * @param createNodeDTO 
     */
    @Put('/edit')
    @ApiQuery({ name: 'id', type: String }) 
    async editNode(
      @Res() res,
      @Query('id', new ValidateObjectId()) id,
      @Body() createNodeDTO: CreateNodeDTO,
    ) {
      if (id != createNodeDTO._id) {
        throw new NotFoundException('Incorrect node id');
      }
      const editedNode= await this.universitiesService.editNode(id, createNodeDTO);
      if (!editedNode) {
          throw new NotFoundException('Node does not exist!');
      }
      return res.status(HttpStatus.OK).json({
        message: 'Node has been successfully updated',
        node: editedNode,
      });
    }
    
    /**
     * Delete a node using id
     * @param res 
     * @param id 
     */
    @Delete('/delete')
    @ApiQuery({ name: 'id', type: String })
    async deleteNode(@Res() res, @Query('id', new ValidateObjectId()) id) {
      const deletedNode = await this.universitiesService.deleteNode(id);
      if (!deletedNode) {
          throw new NotFoundException('Node does not exist!');
      }
      return res.status(HttpStatus.OK).json({
        message: 'Node has been deleted!',
        node: deletedNode,
      });
    }
  }
