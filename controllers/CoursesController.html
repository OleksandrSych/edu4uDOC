    /**
     * Fetch all courses
     * @param res 
     */
    @Get('/courses')
    @ApiOperation({description: `Get courses
      \nIf user role is 'admin' return all couses
      \nElse return couses where user is course owner`
    ,summary: 'Fetch all courses'})
    @ApiOkResponse(
      {description: 'The courses list has been successfully returned.'})
    @ApiForbiddenResponse({description:'Forbidden.'})
  async getUsers(@Res() res) {
    const courses = await this.coursesService
      .getCourses(new ObjectId(res.req.user._id));    
    return res.status(HttpStatus.OK).json(courses);
  }
    
    /**
     * Fetch a particular course using id
     * @param res 
     * @param id 
     */
    @Get('/:id')
    @ApiOperation({description: `Get course by id
      \nIf user role is 'admin' return course
      \nElse return couse where user is course owner`
    ,summary: 'Fetch a particular course using id'})
    @ApiOkResponse({ description:'The course has been successfully returned.'})
    @ApiForbiddenResponse({description:'Forbidden.'})
    @ApiNotFoundResponse({description:'Course doesn\'t exist.'})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({ name: 'id', type: String })
    async getCourse(@Res() res, @Param('id', new ValidateObjectId()) id) {
      const course = await this.coursesService.getCourse(new ObjectId(id));
      const userId = new ObjectId(res.req.user._id);
      const user = await this.usersService.getUser(userId); 
      if (!course || (user.role != 'admin' && !user._id.equals(course._id))) {
        throw new NotFoundException('Course doesn\'t exist!');
      }
      return res.status(HttpStatus.OK).json(course);
    }

    @Get('/mentors/:courseId')
    @ApiOperation({description: 'Get mentors profiles by course Id',
      summary: 'Fetch mentors profiles'})
    @ApiOkResponse({description: `This Request is OK`})
    @ApiForbiddenResponse({description:'Forbidden.'})
    @ApiNotFoundResponse({description:'Mentors profiles do not exist.'})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({ name: 'courseId', type: String })
    async getMentorsCourseId(@Res() res, 
      @Param('courseId', new ValidateObjectId()) id) {
      const course = await this.coursesService.getCourse(new ObjectId(id));

    /**
     * Submit a course
     * @param res 
     * @param createCourseDTO 
     */
    @Post('/add') 
    @ApiOperation({description: 'Add new Course'
      ,summary: 'Add new Course'})
    @ApiOkResponse({description: 'Course has been submitted successfully!'})
    @ApiForbiddenResponse({description: 'It is not a teacher'})
    async addCourse(@Res() res, @Body() createCourseDTO: CreateCourseDTO) { 
      createCourseDTO.createdDate = new Date();
      createCourseDTO.createdUser = new ObjectId(res.req.user._id);
      if(!createCourseDTO.teacherOwner){
        createCourseDTO.teacherOwner = new ObjectId(res.req.user._id);
      } 
      const deps = 
      await this.univerService.getTeacherNode(new ObjectId(res.req.user._id));
      if(deps.length == 0){
        throw new ForbiddenException('It is not a teacher');
      }
      createCourseDTO.parentDepartment = deps[0]._id;       
      const newCourse = await this.coursesService.addCourse(createCourseDTO);
      return res.status(HttpStatus.OK).json({
        message: 'Course has been submitted successfully!',
        course: newCourse,
      });
    }
    
    /**
     * Add new Activity
     * @param res 
     * @param courseId 
     * @param topicId 
     * @param activityDTO 
     */
    @Post('/activity/add') 
    @ApiOperation({description: 'Add new Activity'
      ,summary: 'Add new Activity'})
    @ApiOkResponse({description: 'Course has been submitted successfully!'})
    @ApiForbiddenResponse({description: 'It is not a teacher'})
    @ApiQuery({name: 'courseId', type: String })
    @ApiQuery({name: 'topicId', type: String })
    async addActivity(@Res() res, 
                      @Query('courseId', new ValidateObjectId()) courseId , 
                      @Query('topicId', new ValidateObjectId()) topicId, 
                      @Body() activityDTO: ActivityDTO) { 
      let course = await this.coursesService.getCourse(new ObjectId(courseId));
      if (!course) {
        throw new NotFoundException('Course doesn\'t exist!');
      }
      if (!course.topics) {
        throw new NotFoundException('Course doesn\'t have topics!');
      }
      let topicUpdated = false; 
      course.topics.forEach(function(topic){
        if(!topicUpdated && topic._id.equals(new ObjectId(topicId))){
          if(!topic.activities){
            topic.activities = [];
          }
          topic.activities.push(activityDTO);
          topicUpdated = true;
        }
      });
      if(!topicUpdated){
        throw new NotFoundException('Course doesn\'t have this topic');
      }
      course = await this.coursesService.editCourse(course);
      return res.status(HttpStatus.OK).json({
        message: 'Course has been submitted successfully!',
        course: course,
      });
    }

    /**
     * Add new Activity Link
     * @param res 
     * @param courseId 
     * @param topicId 
     * @param activityDTO 
     */
    @Post('/activity/link/add') 
    @ApiOperation({description: 'Add new Activity Link'
      ,summary: 'Add new Activity Link'})
    @ApiOkResponse({description: 'Course has been submitted successfully!'})
    @ApiForbiddenResponse({description: 'It is not a teacher'})
    @ApiQuery({name: 'courseId', type: String })
    @ApiQuery({name: 'activityId', type: String })
    @ApiQuery({name: 'link', type: String })
    async addActivityLink(@Res() res, 
                      @Query('courseId', new ValidateObjectId()) courseId , 
                      @Query('activityId', new ValidateObjectId()) activityId, 
                      @Query('link') link, ) { 
      let course = await this.coursesService.getCourse(new ObjectId(courseId));
      if (!course) {
        throw new NotFoundException('Course doesn\'t exist!');
      }
      if (!course.topics) {
        throw new NotFoundException('Course doesn\'t have topics!');
      }
      let activityUpdated = false; 
      course.topics.forEach(function(topic){
        if(!activityUpdated && topic.activities){
          topic.activities.forEach(function(act){
            if(act._id == activityId){
              if(!act.links){
                act.links = [];
              }
              act.links.push(link);
              activityUpdated = true;
            }
