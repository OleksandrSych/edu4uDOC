import { Controller, 
  Request, 
  Post, 
  UseGuards, 
  Get, 
  Headers, 
  Body, 
  ForbiddenException,
  NotFoundException,
  Param } from '@nestjs/common';
import { ApiTags,
  ApiUnauthorizedResponse, 
  ApiOkResponse, 
  ApiForbiddenResponse, 
  ApiBearerAuth, 
  ApiNotFoundResponse,
  ApiOperation,
  ApiParam,
  ApiBadRequestResponse} from '@nestjs/swagger'; 
import { AuthGuard } from '@nestjs/passport';
import { AuthService } from './auth/auth.service';
import { JwtAuthGuard } from './auth/jwt-auth.guard';
import { UsersService } from './users/users.service';
import { MailerEdu4uService } from './mailer-edu4u/mailer-edu4u.service';
import { ObjectId } from 'mongodb'; 
import {} from 'jwt-decode';
import { UserLoginDTO } from './users/dto/userilogin.dto';
import { ValidateObjectId } from './pipes/validate-object-id.pipes';

/**
 * App Controller
 */
@ApiTags('Login and Logout')
@Controller()
export class AppController {
  /**@ignore */
  constructor(private authService: AuthService,
              private usersService: UsersService,
              private mailerService: MailerEdu4uService,) {}
  
  /**
   * login User
   * @param req 
   * @param userLoginDto 
   */
  @UseGuards(AuthGuard('local'))
  @Post('/auth/login')
  @ApiOkResponse({description: 'Token and user profile'})
  @ApiForbiddenResponse({description: 'Email doesn\'t confirmed'})
  @ApiUnauthorizedResponse({description: 'User doesn\'t exist or Wrong credentials.'})
  // eslint-disable-next-line
  async login(@Request() req, @Body() userLoginDto: UserLoginDTO) {
    if(req.user['_doc'].role){
      const role = req.user['_doc'].role;
      if((role == 'student' || role == 'mentor') && !req.user['_doc'].confirmedEmail ){
        throw new ForbiddenException('Email doesn\'t confirmed');
      }
    }
    return {
      access_token : (await this.authService.login(req.user['_doc'])).access_token,
      profile: req.user['_doc']
    };
  }

  /**
   * logout User
   * @param req 
   */
  @Post('/auth/logout')
  @ApiOkResponse({description: 'Always return statusCode: 200'})  
  async logout() {
    return {
      statusCode: 200
    }
  }

  /**
   * Refresh token
   * @param headers 
   */
  @Post('/auth/refresh') 
  async refresh( @Headers() headers) { 
    const token = headers.authorization.replace('Bearer ','');
    return this.authService.refresh(token);
  }

  /**
   * Send message to user email with reference how to change password
   * @param userLoginDto 
   */
  @ApiOperation({description: 'Send message to user email with reference how to change password'})
  @ApiNotFoundResponse({description: 'User with this email doesn\'t exist!'})
  @ApiOkResponse({description: 'Message has been sent to user email'})  
  @Post('/auth/password') 
  async changePassword(@Body() userLoginDto: UserLoginDTO) { 
    const user = await this.usersService.findUserByEmail(userLoginDto.username);
    if(!user){
      throw new NotFoundException('User with this email doesn\'t exist!');
    }
    this.mailerService.sendMail(
      user.email, 
      'Запит на зміну паролю EDU4U', 
      this.mailerService.changePassword(user)); 
    return {
      message: `Message has been sent to ${user.email}`
    }
  }

  /**
   * Get User profile by id
   * @param id user id 
   */ 
  @Get('/profile/:id')
  @ApiBadRequestResponse({description: 'Invalid id!'})
  @ApiParam({name: 'id', type: String }) 
  @ApiOkResponse({description: 'Return user profile and access_token'})
  @ApiUnauthorizedResponse({description: 'User Unauthorized'})
  async getProfileById(@Request() req, @Param('id', new ValidateObjectId()) id) {
    const user = await this.usersService.getUser(new ObjectId(id));
    if(!user){
      throw new NotFoundException('User doesn\'t exist!');
    } 
    return {
      access_token: (await this.authService.login(user)).access_token,
      profile: user
    }
  }
   
  /**
   * Get User profile
   * @param req 
   */
  @UseGuards(JwtAuthGuard)
  @Get('/profile')
  @ApiBearerAuth()  
  @ApiOkResponse({description: 'Return user profile'})
  @ApiUnauthorizedResponse({description: 'User Unauthorized'})
  async getProfile(@Request() req) {
    return await this.usersService.getUser(new ObjectId(req.user._id));
  }
}
