import { Body, Controller, HttpStatus, Post, Res, UseGuards } from '@nestjs/common';
import { ApiBearerAuth, ApiTags, ApiUnauthorizedResponse } from '@nestjs/swagger';
import { JwtAuthGuard } from 'src/auth/jwt-auth.guard';
import { InviteUserDTO } from './dto/invite-user.dto';
import { InviteUserService} from './invite-user.service';

/**
 * Invite Users
 */
@UseGuards(JwtAuthGuard)
@Controller('invite')
@ApiTags('Invite Users')
@ApiBearerAuth()
@ApiUnauthorizedResponse({description:'Access Token is not valid or has expired'})
export class InviteUserController {
    /**@ignore */
    constructor(private inviteUserService: InviteUserService) { } 

    /**
     * Invite Techer
     * @param res 
     * @param inviteUserDTO 
     */
    @Post('/teacher')
    async inviteTecher(@Res() res, @Body() inviteUserDTO: InviteUserDTO) {
        const newTeacher = await this.inviteUserService.inviteTecher(inviteUserDTO);
        return res.status(HttpStatus.OK).json({
          message: 'Teacher has been invited successfully!',
          user: newTeacher,
        });
    }
    
   
}
