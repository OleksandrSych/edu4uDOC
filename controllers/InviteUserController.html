import { Body, 
  Controller, 
  ForbiddenException, 
  Get, 
  HttpStatus, 
  Param, 
  Post, 
  Res, 
  UseGuards } from '@nestjs/common';
import { ApiBadRequestResponse, ApiBearerAuth, 
  ApiForbiddenResponse, 
  ApiNotFoundResponse, 
  ApiOkResponse, 
  ApiOperation, 
  ApiParam, 
  ApiTags, 
  ApiUnauthorizedResponse } from '@nestjs/swagger';
import { ObjectId } from 'mongodb';
import { JwtAuthGuard } from 'src/auth/jwt-auth.guard';
import { ValidateObjectId } from 'src/pipes/validate-object-id.pipes';
import { CreateUserDTO } from 'src/users/dto/create-user.dto';
import { InviteUserDTO } from './dto/invite-user.dto';
import { InviteUserService} from './invite-user.service';
import { UsersService } from '../users/users.service';

/**
 * Invite Users
 */

@Controller('invite')
@ApiTags('Invite Users')
export class InviteUserController {
  /**@ignore */
  constructor(private inviteUserService: InviteUserService,
              private usersService: UsersService) { } 

    /**
     * Invite Teacher
     * @param res 
     * @param inviteUserDTO 
     */
    @Post('/teacher')
    @UseGuards(JwtAuthGuard)
    @ApiBearerAuth()
    @ApiUnauthorizedResponse({description: 'Access Token is not valid or has expired'})
    @ApiOkResponse({description: 'New teacher profile'})
    @ApiOperation({description: `Invite teacher
        \nSend invitation to new teacher
        \nWaiting for the department id and the teacher's email`})
  async inviteTeacher(@Res() res, @Body() inviteUserDTO: InviteUserDTO) {
    const newTeacher = 
      await this.inviteUserService.inviteTeacher(inviteUserDTO);
    return res.status(HttpStatus.OK).json({
      message: 'Teacher has been invited successfully!',
      user: newTeacher,
    });
  }
    
    /**
     * Invite Mentor
     * @param res 
     * @param createUserDTO 
     */
    @Post('/mentor/:inviteKey')
    @ApiForbiddenResponse({description:' User with this email already exists'})
    @ApiOkResponse({description: 'New mentore profile'})
    @ApiOperation({description: `Invite mentor
        \nRegistry new mentor to the course
        \nWaiting for the invite key and new user profile
       `})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({name: 'inviteKey', type: String})
    async inviteMentor(@Res() res, @Body() createUserDTO: CreateUserDTO, @Param('inviteKey', new ValidateObjectId()) inviteKey) {
      const newMentor = await this.inviteUserService.inviteUser( new ObjectId(inviteKey), createUserDTO, 'mentor');
      return res.status(HttpStatus.OK).json({
        message: 'Mentor has been invited successfully!',
        user: newMentor,
      });
    }

    /**
     * Invite Student
     * @param res 
     * @param createUserDTO 
     */
    @Post('/student/:inviteKey')    
    @ApiForbiddenResponse({description: 'User with this email already exists'})
    @ApiOkResponse({description: 'New student profile'})
    @ApiOperation({description: `Invite student
        \nRegistry new student to the course
        \nWaiting for the invite key and new user profile
       `})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({name: 'inviteKey', type: String})
    async inviteStudent(@Res() res, @Body() createUserDTO: CreateUserDTO, @Param('inviteKey', new ValidateObjectId()) inviteKey) {
      const newStudent= await this.inviteUserService.inviteUser(new ObjectId(inviteKey), createUserDTO, 'student');
      return res.status(HttpStatus.OK).json({
        message: 'Student has been invited successfully!',
        user: newStudent,
      });
    }

    /**
     * Confirm Email
     * @param res 
     * @param id user Id
     */
    @Post('/confirm/:id')    
    @ApiForbiddenResponse({description: 'User with this key is not exists'})
    @ApiOkResponse({description: 'Email confirmed'})
    @ApiOperation({description: 'Confirm user or mentor Email'})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({name: 'id', type: String})
    async confirmEmail(@Res() res, @Param('id', new ValidateObjectId()) id) {
      const user = await this.usersService.getUser(new ObjectId(id));
      if(!user){
        throw new ForbiddenException('User with this key is not exists');
      }
      user.confirmedEmail = true;
      await this.usersService.editUser(new ObjectId(id), user);
      return res.status(HttpStatus.OK).json({
        message: 'Email has been confirmed successfully!'
      });
    }

    /**
     * Check cpurse invite key
     * @param res 
     * @param inviteKey 
     */
    @Get('/check/:inviteKey')    
    @ApiNotFoundResponse({description: 'Not found course with this inviteKey'})
    @ApiOkResponse({description: 'Course id and course name'})
    @ApiOperation({description: `Check invite key 
        \nReturn course id and course name
       `})
    @ApiBadRequestResponse({description:'Invalid id!'})
    @ApiParam({name: 'inviteKey', type: String})
    async checkInviteKey(@Res() res, 
      @Param('inviteKey', new ValidateObjectId()) inviteKey) {
      const id = new ObjectId(inviteKey);
      const course = await this.inviteUserService.checkInviteKey(id);
      return res.status(HttpStatus.OK).json({
        id: course._id,
        courseName: course.courseName});
    }
   
}
