import { Body, Controller, ForbiddenException, Get, HttpStatus, NotFoundException, Param, Post, Res } from '@nestjs/common';
import { ApiForbiddenResponse, ApiNotFoundResponse, ApiOkResponse, ApiParam, ApiTags, ApiUnauthorizedResponse } from '@nestjs/swagger';
import { ValidateObjectId } from 'src/pipes/validate-object-id.pipes';
import { TeachersService} from './teachers.service'
import { UsersService } from '../users/users.service';
import { AuthService} from '../auth/auth.service';
import { CreateUserDTO } from 'src/users/dto/create-user.dto';


/**
 * Teachers Controller
 */
@ApiTags('Teachers')
@Controller('teachers')
export class TeachersController {
    /**@ignore */
    constructor(private teachersService: TeachersService,
        private usersService: UsersService,
        private authService: AuthService) { }  

    /**
     * Get new teacher by id
     * @param res 
     * @param id 
     */
    @Get('/newteacher/:id')
    @ApiParam({ name: 'id', type: String })
    @ApiOkResponse({description: 'result new teacher email'})
    @ApiForbiddenResponse({description: 'Teacher does not new!'})
    @ApiUnauthorizedResponse({description: 'Teacher doesn\'t exist!'})
    @ApiNotFoundResponse({description: 'Teacher doesn\'t exist!'})
    async loginNewUser(
        @Res() res, @Param('id', new ValidateObjectId()) id) {
        const teacher = await this.teachersService.getNewTeacher(id);   
        var result = { 'teacher_email' : teacher.email};
        return res.status(HttpStatus.OK).json(result);     
    }    

    /**
     * Submit a new teacher
     * @param res 
     * @param createUserDTO 
     */    
    @Post('/add')
    @ApiForbiddenResponse({description:'User with this email already exists'})
    async addUser(@Res() res, @Body() createUserDTO: CreateUserDTO) {
        const teacher = await this.teachersService.getNewTeacher(createUserDTO._id); 
        if(teacher.email != createUserDTO.email){
            throw new ForbiddenException('Teacher has wrong email!');
        }
        createUserDTO.role = 'teacher';
        var user = await this.usersService.editUser(createUserDTO._id , createUserDTO); 
        return res.status(HttpStatus.OK).json( {
            access_token : (await this.authService.login(user)).access_token,
            profile: user
          });        
    } 

    
}
