    /**
     * Invite Techer
     * @param inviteUserDTO invite User DTO
     */
    async inviteTecher(inviteUserDTO: InviteUserDTO) {
        if(!inviteUserDTO){
            throw new NotFoundException('Invalid inviteUserDTO');
        }  
        var node = await  this.universitiesService.getNode(inviteUserDTO.nodeId);
        if(!node){
            throw new NotFoundException('Node not found');
        }
        var user = await this.usersService.findUserByEmail(inviteUserDTO.email);
        if(!user){
            var newUser = new CreateUserDTO;
            newUser.email = inviteUserDTO.email;
            newUser.role = 'teacher';
            newUser.password = 'newUserPassword';
            newUser.contactPhone = '';
            newUser.firstName = '';
            newUser.lastName = '';
            newUser.title = '';
            newUser.interests = '';
            user = await this.usersService.addUser(newUser);
        }
        if(user.role != 'teacher'){ 
          user = await this.usersService.editUser(user._id, user);
        }          
        if(!node.teachers.includes(user._id))
            node.teachers.push(user._id);
        this.universitiesService.editNode(node._id, node);
        var institite = await this.universitiesService.getNode(node.parentNode);
        var univer = await this.universitiesService.getNode(institite.parentNode); 
        if(!inviteUserDTO.homePage){
          inviteUserDTO.homePage = 'https://edu4u.gitlab.io/edu4u/#'
        }
        this.mailerEdu4uService.sendMail(
          user.email, 
          'Реєстрація викладача EDU4U', 
          this.getMailText(node.nodeName, institite.nodeName, univer.nodeName, inviteUserDTO, user ));
        return user;
    }
