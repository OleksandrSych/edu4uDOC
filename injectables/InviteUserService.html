import { ForbiddenException, Injectable, NotFoundException } from '@nestjs/common';
import { InviteUserDTO } from './dto/invite-user.dto';
import { UsersService } from '../users/users.service';
import { CoursesService} from '../courses/courses.service'
import { UniversitiesService } from '../universities/universities.service';
import { MailerEdu4uService } from '../mailer-edu4u/mailer-edu4u.service';
import { CreateUserDTO } from 'src/users/dto/create-user.dto';
import { ObjectId } from 'mongodb'; 
import { Course } from 'src/courses/interfaces/course.interface'; 

/**
 * Invite User Service
 */
@Injectable()
export class InviteUserService {
  /**@ignore */
  constructor(private usersService: UsersService, 
                private universitiesService: UniversitiesService,
                private mailerEdu4uService: MailerEdu4uService, 
                private coursesService: CoursesService) { }
  /**
  * Invite Teacher
  * @param inviteUserDTO invite User DTO
  */
  async inviteTeacher(inviteUserDTO: InviteUserDTO) {
    if(!inviteUserDTO){
      throw new NotFoundException('Invalid inviteUserDTO');
    }
    const id = new ObjectId(inviteUserDTO.nodeId);
    const node = await this.universitiesService.getNode(id);
    if(!node){
      throw new NotFoundException('Node not found');
    }
    let user = await this.usersService.findUserByEmail(inviteUserDTO.email);
    if(!user){
      const newUser = new CreateUserDTO;
      newUser.email = inviteUserDTO.email;
      newUser.role = 'teacher';
      newUser.password = 'newUserPassword';
      newUser.contactPhone = '';
      newUser.firstName = '';
      newUser.lastName = '';
      newUser.title = '';
      newUser.interests = '';
      user = await this.usersService.addUser(newUser);
    }
    if(user.role != 'teacher'){ 
      user = await this.usersService.editUser(user._id, user);
    }          
    if(!node.teachers.includes(user._id))
      node.teachers.push(user._id);
    this.universitiesService.editNode(node._id, node);
    const institite = 
      await this.universitiesService.getNode(node.parentNode);
    const univer = 
      await this.universitiesService.getNode(institite.parentNode); 
    this.mailerEdu4uService.sendMail(
      user.email, 
      'Реєстрація викладача EDU4U', 
      this.mailerEdu4uService.inviteTeacher(node.nodeName, 
        institite.nodeName, 
        univer.nodeName, 
        user ));
    return user;
  }

  /**
  * Invite User(Mentor or Student)
  * @param inviteUserDTO invite User DTO
  */
  async inviteUser(inviteKey: ObjectId, 
    createUserDTO: CreateUserDTO, 
    newRole: string) {
    if(!createUserDTO){
      throw new NotFoundException('Invalid createUserDTO');
    }
    const course = await this.checkInviteKey(inviteKey); 
    let isOldUser = true;
    let user = await this.usersService.findUserByEmail(createUserDTO.email);
    if(!user){
      createUserDTO.role = newRole;
      user = await this.usersService.addUser(createUserDTO);
      isOldUser = false;
    } 
    if(newRole == 'mentor'){
      //push new mentor if not exists
      (!course.mentors.includes(user._id)) && course.mentors.push(user._id);
      await this.coursesService.editCourse(course); 
      this.mailerEdu4uService.sendMail(
        user.email, 
        'Реєстрація ментора EDU4U', 
        this.mailerEdu4uService.inviteMentor(course.courseName, user));
    }
    else if (!course.students.includes(user._id)) {
      //push new student if not exists
      (!course.students.includes(user._id)) && course.students.push(user._id);
      await this.coursesService.editCourse(course);      
      this.mailerEdu4uService.sendMail(
        user.email, 
        'Реєстрація студента EDU4U', 
        this.mailerEdu4uService.inviteStudent(course.courseName, user));
    }
    if(isOldUser)
      throw new ForbiddenException('User with this email already exists');
  }

  /**
   * Check Invite Key
   * @param inviteKey invite Key
   */
  async checkInviteKey(inviteKey: ObjectId):Promise<Course> {
    const courses = await this.coursesService.getCoursesInviteKey(inviteKey); 
    if(courses.length == 0 ){
      throw new NotFoundException('Not found course with this inviteKey');
    }
    return courses[0];      
  }
