  /**
     * Send Mail
     * @param mailTo sendMail
     * @param subject mailsubject
     * @param mailText mail Text
     */
  public sendMail(mailTo: string, subject: string, mailText: string): void {
    this.mailerService
      .sendMail({
        to: mailTo,
        from: 'edu4u.user@gmail.com', 
        subject: subject, 
        html: mailText
      });
  }
