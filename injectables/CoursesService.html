import { Injectable } from '@nestjs/common';
import { Model } from 'mongoose';
import { Course } from './interfaces/course.interface';
import { InjectModel } from '@nestjs/mongoose';
import { ObjectId } from 'mongodb';
import { CreateCourseDTO } from './dto/create-course.dto';
import { UsersService } from '../users/users.service';

/**
 * Courses Service
 */
@Injectable()
export class CoursesService { 
  /**@ignore */
  constructor(@InjectModel('Course') private readonly courseModel: Model<Course>,
              private usersService: UsersService, ) { }
    
  /**
     * Get all users courses
     */
  async getCourses(id: ObjectId) : Promise<Course[]> {
    let courses = await this.courseModel.find().exec();
    const user = 
      await this.usersService.getUser(id);
    if(user.role != 'admin'){
      courses = courses.filter(course => course.teacherOwner.equals(user._id));
      courses.forEach(function(course){
        course.teacherName = user.firstName +' '+ user.lastName;
      });
    } 
    else{
      const teachers = await this.usersService.getTeachers();
      courses.forEach(function(course){
        const teacher = teachers.find(
          teacher => course.teacherOwner.equals(teacher._id));
        if(teacher) {
          course.teacherName = teacher.firstName +' '+ teacher.lastName;
        } 
      });
    }
    return courses;
  }

  /**
   * Get all mentors 
   * @param id User Id
   */
  async getMentors(id: ObjectId) {
    const courses = await this.getCourses(id);
    const users = await this.usersService.getUsers();
    const mentors = [];
    courses.map(course =>{
      if(course.mentors){
        course.mentors.map(mentor=>{
          const user = users.find(u=>u._id.equals(mentor))
          if(user){
            mentors.push(
              {
                courseId : course._id,
                courseName : course.courseName,
                userId : user._id,
                userFirstName : user.firstName,
                userLastName : user.lastName,
              }
            );
          }
        })
      }
    });
    return mentors;
  }

  /**
   * Get all students 
   * @param id User Id
   */
  async getStudents(id: ObjectId) {
    const courses = await this.getCourses(id);
    const users = await this.usersService.getUsers();
    const students = [];
    courses.map(course =>{
      if(course.students){
        course.students.map(student=>{
          const user = users.find(u=>u._id.equals(student))
          if(user){
            students.push(
              {
                courseId : course._id,
                courseName : course.courseName,
                userId : user._id,
                userFirstName : user.firstName,
                userLastName : user.lastName,
              }
            );
          }
        })
      }
    });
    return students;
  }

  /**
     * Get course by id
     * @param id Course id
     */
  async getCourse(id: ObjectId): Promise<Course> { 
    const course = await this.courseModel
      .findOne(id)
      .exec(); 
    return course;
  }
  
  /**
     * Get course by id
     * @param id Course id
     */
  async updateCourseInviteKey(id: ObjectId): Promise<Course> { 
    const course = await this.courseModel
      .findOne(id)
      .exec();
    if(!course){
      return;
    }
    course.inviteKey = new ObjectId();
    const editedCourse= await this.courseModel
      .findByIdAndUpdate(course.id, course)
      .exec();
    return editedCourse; 
  }
    
  /**
     * Get courses by parent department id
     * @param id parent department id
     */
  async getCoursesDepartmentId(id: ObjectId): Promise<Course[]> {
    const courses = await this.courseModel
      .find({"parentDepartment" : id})
      .exec();
    return courses;
  }

  /**
     * Get courses by inviteKey
     * @param inviteKey inviteKey
     */
  async getCoursesInviteKey(inviteKey: ObjectId): Promise<Course[]> {
    const courses = await this.courseModel
      .find({"inviteKey" : inviteKey})
      .exec();
    return courses;
  }

  /**
     * Add new course
     * @param createCourseDTO 
     */
  async addCourse(createCourseDTO: CreateCourseDTO): Promise<Course> {
    createCourseDTO._id = new ObjectId();
    createCourseDTO.inviteKey = new ObjectId();
    if(createCourseDTO.topics){
      createCourseDTO.topics.forEach(function(topic){
        topic._id = new ObjectId();
        if(topic.activities){
          topic.activities.forEach(function(activity){
            activity._id = new ObjectId();
          });
        }         
      });
    }
    const newCourse = await new this.courseModel(createCourseDTO);
    return newCourse.save();
  }

  /**
     * Edit course data 
     * @param createCourseDTO 
     */
  async editCourse(createCourseDTO: CreateCourseDTO) : Promise<Course> {
    if(createCourseDTO.topics){
      createCourseDTO.topics.forEach(function(topic){
        if(!topic._id){
          topic._id = new ObjectId();
        }
        if(topic.activities){
          topic.activities.forEach(function(activity){
            if(!activity._id){
              activity._id = new ObjectId();
            }
          });
        }         
      });
    }
    const editedCourse= await this.courseModel
      .findByIdAndUpdate(createCourseDTO._id, createCourseDTO, { new: true });
    return editedCourse;
  }

  /**
     * Delete course
     * @param id course id
     */
  async deleteCourse(id: ObjectId): Promise<Course> {
    const deletedCourse = await this.courseModel
      .findByIdAndRemove(id);
    return deletedCourse;
  }    
}
