import { Injectable } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';
import { Node } from './interfaces/node.interface';
import { ObjectId } from 'mongodb';
import { CreateNodeDTO } from './dto/create-node.dto';
 
import { connectableObservableDescriptor } from 'rxjs/internal/observable/ConnectableObservable';
import { UsersService } from 'src/users/users.service';

/**
 * Universities Service
 */
@Injectable()
export class UniversitiesService {
 
    /**
     * @ignore
     * @param nodeModel 
     */
    constructor( @InjectModel('Node') private readonly nodeModel: Model<Node>, private usersService: UsersService ) { }
    
    /**
     * Get Nodes by parentNode id
     * @param id 
     */
    async getNodes(id: string) : Promise<Node[]> {
        const nodes = await this.nodeModel
            .find({"parentNode" : id})
            .exec();
        return nodes;
    }

    /**
     * Get Node by id
     * @param id 
     */
    async getNode(id): Promise<Node> { 
        const node = await this.nodeModel
            .findOne({"_id": id})
            .exec(); 
        return node;
    }

    /**
     * Get all tree of all nodes
     */
    async getTree() : Promise<string[]> {
        var allUsers = await this.usersService.getUsers();
        var allNodes = await this.nodeModel
            .find({}) 
            .exec();
        var nodes = allNodes.filter(x=>x.parentNode =="");
        var tree =[];
        nodes.map(element => 
               tree.push( {
                "id" : element._id,
                "name" : element.nodeName,  
                "institutions" : this.getInstitutions(element._id,allNodes,allUsers)                    
            }));
        return tree
    }

    /**
     * get Institutions
     * @param id universiry id
     * @param allNodes all nodes
     * @param allUsers all users
     */
    getInstitutions(id, allNodes,allUsers) {
        var nodes = allNodes.filter(x=>x.parentNode == id);
        var tree =[];
        (nodes).map(node=>{
            tree.push( {
                "id" : node._id,
                "name" : node.nodeName,
                "departments": this.getDepartments(node["id"], allNodes, allUsers)                              
            })});      
        return tree;
    }
  
    /**
     * get Departments
     * @param id Institute id
     * @param allNodes all nodes
     * @param allUsers all nodes
     */
    getDepartments(id, allNodes, allUsers) {
        var nodes = allNodes.filter(x=>x.parentNode == id);
        var tree =[];
        (nodes).map(node=>{
            tree.push( {
                "id" : node._id,
                "name" : node.nodeName ,
                "teachers" : this.getTeachers(node.teachers, allUsers)                     
            })});  
        return tree;
    }
  
    /**
     * get Teachers
     * @param teachers 
     * @param allUsers 
     */
    getTeachers(teachers, allUsers) {
        if(teachers.length == 0 )
            return [];
        var tree =[];
        (teachers).map(teacher=>{      
            var user = allUsers.filter(x=>x._id == teacher)[0];
            if(user){
              tree.push( {
                  "id" : teacher,
                  "name" : user.lastName + " " + user.firstName,
                  "email" : user.email
              })}          
          }); 
        return tree;
    }


    /**
     * Add new Node
     * @param createNodeDTO 
     */
    async addNode(createNodeDTO: CreateNodeDTO): Promise<Node> {
        createNodeDTO._id = new ObjectId();
        const newNode = await new this.nodeModel(createNodeDTO);
        return newNode.save();
    }

    /**
     * Edit Node by id
     * @param id 
     * @param createNodeDTO 
     */
    async editNode(id: any, createNodeDTO: CreateNodeDTO) : Promise<Node> {
        const editedNode = await this.nodeModel
            .findByIdAndUpdate(id, createNodeDTO, { new: true });
        return editedNode;
    }

    /**
     * Delete Node by id
     * @param id 
     */
    async deleteNode(id): Promise<any> {
        const deletedNode = await this.nodeModel
            .findByIdAndRemove(id);
        return deletedNode;
    }
}
