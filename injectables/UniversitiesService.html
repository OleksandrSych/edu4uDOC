import { Injectable } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';
import { Node } from './interfaces/node.interface';
import { User } from '../users/interfaces/user.interface';
import { ObjectId } from 'mongodb';
import { CreateNodeDTO } from './dto/create-node.dto';
import { UsersService } from 'src/users/users.service';

/**
 * Universities Service
 */
@Injectable()
export class UniversitiesService {
 
  /**
     * @ignore
     * @param nodeModel 
     */
  constructor( @InjectModel('Node') private readonly nodeModel: Model<Node>, private usersService: UsersService ) { }
    
  /**
     * Get Nodes by parentNode id
     * @param id 
     */
  async getNodes(id: ObjectId) : Promise<Node[]> {
    const nodes = await this.nodeModel
      .find({"parentNode" : id})
      .exec();
    return nodes;
  }

  /**
     * Get Node by id
     * @param id 
     */
  async getNode(id: ObjectId): Promise<Node> { 
    const node = await this.nodeModel
      .findOne(id)
      .exec(); 
    return node;
  }

  /**
     * Get informations about all teachers
     */
  async getTeachersInfo(): Promise<string[]> {
    const allUsers = await this.usersService.getUsers();
    const allNodes = await this.nodeModel
      .find({}) 
      .exec();
    const nodes = allNodes.filter(x=>x.teachers.length > 0);
    const teachersInfo =[];
    nodes.map(department => {
      const institute = 
            allNodes.filter(x=>
              department.parentNode && x._id.equals(department.parentNode))[0];
      if(institute){    
        const university = 
            allNodes.filter(x=>
              institute.parentNode && x._id.equals(institute.parentNode))[0];
        if(university){
          department.teachers.map(async teacher =>{
            const user = await allUsers.filter(x=>x._id.equals(teacher))[0];
            if(user){ 
              teachersInfo.push( {
                "teacher" : { "id" : teacher, "firstName": user.firstName, "lastName": user.lastName},
                "university" : {"id": university._id , "name":university.nodeName},  
                "institute" : {"id": institute._id , "name":institute.nodeName},
                "department" : {"id": department._id , "name":department.nodeName},
                "courses" : 0
              }
              )}})}}});
    return teachersInfo;
  }

  /**
     * Get nodes by teacher Id
     */
  async getTeacherNode(id: ObjectId): Promise<Node[]> {
    const allNodes = await this.nodeModel
      .find({}) 
      .exec();
    return allNodes.filter(x=>x.teachers.includes(id));      
  }

  /**
     * Get all tree of all nodes
     */
  async getTree() : Promise<string[]> {
    const allUsers = await this.usersService.getUsers();
    const allNodes = await this.nodeModel
      .find({}) 
      .exec();
    const nodes = allNodes.filter(x=>x.parentNode === null);
    const tree =[];
    nodes.map(element => 
      tree.push( {
        "id" : element._id,
        "name" : element.nodeName,  
        "institutions" : this.getInstitutions(element._id,allNodes,allUsers)                    
      }));
    return tree
  }

  /**
     * get Institutions
     * @param id universiry id
     * @param allNodes all nodes
     * @param allUsers all users
     */
  getInstitutions(id: ObjectId, allNodes: Node[],allUsers: User[]) {
    const nodes = allNodes.filter(x=>x.parentNode && x.parentNode.equals(id));
    const tree =[];
    (nodes).map(node=>{
      tree.push( {
        "id" : node._id,
        "name" : node.nodeName,
        "departments": this.getDepartments(node["id"], allNodes, allUsers)                              
      })});      
    return tree;
  }
  
  /**
     * get Departments
     * @param id Institute id
     * @param allNodes all nodes
     * @param allUsers all nodes
     */
  getDepartments(id: ObjectId, allNodes: Node[], allUsers: User[]) {
    const nodes = allNodes.filter(x=>x.parentNode && x.parentNode.equals(id));
    const tree =[];
    (nodes).map(node=>{
      tree.push( {
        "id" : node._id,
        "name" : node.nodeName ,
        "teachers" : this.getTeachers(node.teachers, allUsers)                     
      })});  
    return tree;
  }
  
  /**
     * get Teachers
     * @param teachers 
     * @param allUsers 
     */
  getTeachers(teachers: ObjectId[], allUsers: User[]) {
    if(teachers.length == 0 )
      return [];
    const tree =[];
    (teachers).map(teacher=>{      
      const user = allUsers.filter(x=>x._id.equals(teacher))[0];
      if(user){
        tree.push( {
          "id" : teacher,
          "name" : user.lastName + " " + user.firstName,
          "email" : user.email
        })}          
    }); 
    return tree;
  }


  /**
     * Add new Node
     * @param createNodeDTO 
     */
  async addNode(createNodeDTO: CreateNodeDTO): Promise<Node> {
    createNodeDTO._id = new ObjectId(); 
    if(!createNodeDTO.parentNode){
      createNodeDTO.parentNode = null;
    }
    const newNode = await new this.nodeModel(createNodeDTO);
    return newNode.save();
  }

  /**
     * Edit Node by id
     * @param id 
     * @param createNodeDTO 
     */
  async editNode(id: ObjectId, createNodeDTO: CreateNodeDTO) : Promise<Node> {
    const oldNode = await this.getNode(id);
    if(!createNodeDTO.parentNode){
      createNodeDTO.parentNode = oldNode.parentNode;
    }
    if(!createNodeDTO.teachers){
      createNodeDTO.teachers = oldNode.teachers;
    }
    if(!createNodeDTO.mentors){
      createNodeDTO.mentors = oldNode.mentors;
    }
    const editedNode = await this.nodeModel
      .findByIdAndUpdate(id, createNodeDTO, { new: true });
    return editedNode;
  }

  /**
     * Delete Node by id
     * @param id 
     */
  async deleteNode(id: ObjectId): Promise<Node> {
    const deletedNode = await this.nodeModel
      .findByIdAndRemove(id);
    return deletedNode;
  }
}
